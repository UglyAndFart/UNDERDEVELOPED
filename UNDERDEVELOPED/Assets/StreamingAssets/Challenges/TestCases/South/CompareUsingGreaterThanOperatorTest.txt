        public static void PlayerCodeTestChecker()
        {
            PlayerCodeTest playeCodeTest = new PlayerCodeTest();
            
            int[,] inputs =
            {
                { 5, 10 },
                { 15, 9 },
                { 9, 14 },
                { 3, 13 },
                { 2, 20 },
                { 56, 45 },
                { 4, 10 },
                { 7, 6 },
                { 11, 21 },
                { 12, 14 }
            };

            bool[] outputs =
            {
                false, true, false, false, false, true, false, true, false, false
            };

            for(int i = 0; i < inputs.GetLength(0); i++)
            {
                bool result = playeCodeTest.CompareUsingGreaterThanOperator(inputs[i, 0], inputs[i, 1]);
                
                if (result == outputs[i])
                {
                    Console.WriteLine("Testcase Passed!");
                }

                else
                {
                    Console.WriteLine("Incorrect Testcase at no." + (i + 1));
                }
            }
